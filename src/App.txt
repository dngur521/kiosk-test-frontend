// src/App.jsx
import { useRef, useState } from "react";
import "./App.css"; // 기본 스타일 유지

// 🚨 중요: 5800H 서버의 DuckDNS 주소로 변경하세요! (wss://.../ws/voice)
const WS_URL = "wss://kemini-kiosk-api.duckdns.org/ws/voice";

function App() {
  const [status, setStatus] = useState("Disconnected");
  const [isRecording, setIsRecording] = useState(false);
  const [transcript, setTranscript] = useState(""); // 서버에서 받은 인식 결과

  const socketRef = useRef(null); // 웹소켓 세션 유지
  const audioContextRef = useRef(null); // 브라우저 오디오 컨텍스트
  const streamRef = useRef(null); // 마이크 스트림

  // 1. 웹소켓 연결
  const connectWebSocket = () => {
    socketRef.current = new WebSocket(WS_URL);

    socketRef.current.onopen = () => {
      setStatus("✅ 서버와 연결되었습니다!");
    };

    socketRef.current.onmessage = (event) => {
      // 구글 STT가 보내준 한국어 텍스트 수신
      console.log("📩 인식 결과 수신:", event.data);
      setTranscript(event.data);
    };

    socketRef.current.onerror = (error) => {
      console.error("❌ 에러 발생:", error);
      setStatus("Error: 연결 실패");
    };

    socketRef.current.onclose = () => {
      console.log("🔒 연결 종료");
      setStatus("연결 종료됨");
    };
  };

  // 2. 마이크 녹음 시작 (16,000Hz LINEAR16 스트리밍)
  const startRecording = async () => {
    if (socketRef.current?.readyState !== WebSocket.OPEN) {
      alert("먼저 서버에 연결해 주세요!");
      return;
    }

    try {
      // 마이크 권한 요청 및 스트림 획득
      streamRef.current = await navigator.mediaDevices.getUserMedia({
        audio: true,
        video: false,
      });

      // 3. 🚨 구글 STT 포맷으로 변환하는 핵심 로직
      // 오디오 컨텍스트를 생성하면서 샘플링 레이트를 16,000Hz로 고정합니다.
      audioContextRef.current = new (
        window.AudioContext || window.webkitAudioContext
      )({
        sampleRate: 16000,
      });

      const source = audioContextRef.current.createMediaStreamSource(
        streamRef.current,
      );

      // AudioProcessor 스크립트 로드
      await audioContextRef.current.audioWorklet.addModule(
        "/src/AudioProcessor.js",
      );
      const workletNode = new AudioWorkletNode(
        audioContextRef.current,
        "audio-processor",
      );

      // 변환된 PCM 데이터를 수신해서 웹소켓으로 전송
      workletNode.port.onmessage = (event) => {
        const pcmBuffer = event.data; // Int16Array의 ArrayBuffer
        if (socketRef.current.readyState === WebSocket.OPEN) {
          socketRef.current.send(pcmBuffer); // 서버로 실시간 전송!
        }
      };

      // 노드 연결
      source.connect(workletNode);
      workletNode.connect(audioContextRef.current.destination);

      setIsRecording(true);
      setTranscript(""); // 결과창 초기화
    } catch (error) {
      console.error("❌ 마이크 캡처 실패:", error);
      alert("마이크 권한을 확인해 주세요.");
    }
  };

  // 4. 녹음 종료
  const stopRecording = () => {
    // 오디오 컨텍스트 닫기
    if (audioContextRef.current) {
      audioContextRef.current.close();
    }
    // 마이크 스트림 끄기
    if (streamRef.current) {
      streamRef.current.getTracks().forEach((track) => track.stop());
    }
    setIsRecording(false);
  };

  return (
    <div className="App">
      <header className="App-header">
        <h1>🎙️ KEMINI Kiosk Voice Test</h1>
        <div
          style={{
            margin: "20px",
            padding: "10px",
            border: "1px solid #ccc",
            background: "#f9f9f9",
          }}
        >
          <p>
            상태: <strong>{status}</strong>
          </p>
        </div>

        <div style={{ display: "flex", gap: "10px", justifyContent: "center" }}>
          <button
            onClick={connectWebSocket}
            disabled={socketRef.current?.readyState === WebSocket.OPEN}
          >
            1. 서버 연결 (wss://...)
          </button>
          <button
            onClick={startRecording}
            disabled={
              !socketRef.current?.readyState === WebSocket.OPEN || isRecording
            }
            style={{ background: "green", color: "white" }}
          >
            2. 녹음 시작
          </button>
          <button
            onClick={stopRecording}
            disabled={!isRecording}
            style={{ background: "red", color: "white" }}
          >
            3. 녹음 종료
          </button>
        </div>

        <div
          style={{
            marginTop: "30px",
            padding: "20px",
            width: "80%",
            border: "2px solid blue",
            minHeight: "100px",
          }}
        >
          <h3>📩 인식 결과 (STT):</h3>
          <p style={{ fontSize: "1.2rem", color: "blue" }}>
            {transcript || "여기에 어르신의 목소리가 텍스트로 나타납니다."}
          </p>
        </div>
      </header>
    </div>
  );
}

export default App;
